# ============================================
# Файл конфігурації .env для Local Brain
# ============================================
# 
# Інструкція:
# 1. Скопіюйте цей файл та перейменуйте на .env
# 2. Замініть your_api_key_here на ваші реальні API ключі
# 3. Розташуйте файл .env поруч з LocalBrain.exe
#
# ============================================

# AI Provider Configuration
# Виберіть провайдера: google_ai або openai
AI_PROVIDER=google_ai

# Google AI Studio API Key
# Отримати ключ: https://aistudio.google.com/app/apikey
GOOGLE_API_KEY=your_google_api_key_here

# OpenAI API Key
# Отримати ключ: https://platform.openai.com/api-keys
OPENAI_API_KEY=your_openai_api_key_here

# OpenAI Base URL (опціонально)
# Використовується для кастомних OpenAI-сумісних API (наприклад, локальні сервери)
# За замовчуванням: https://api.openai.com/v1
OPENAI_BASE_URL=https://api.openai.com/v1

# ============================================
# Приклади конфігурації:
# ============================================
#
# Варіант 1: Тільки Google AI
# AI_PROVIDER=google_ai
# GOOGLE_API_KEY=AIzaSy...
# 
# Варіант 2: Тільки OpenAI
# AI_PROVIDER=openai
# OPENAI_API_KEY=sk-...
# 
# Варіант 3: Обидва провайдери (можна перемикати через інтерфейс)
